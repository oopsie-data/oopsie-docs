 16def write_mp4(video_path: Path, frames: np.ndarray, fps: float) -> None:
 17    """Write RGB frames to an MP4 file.
 18
 19    Args:
 20        video_path (Path): Destination path for the MP4 file.
 21        frames (np.ndarray): Video frames with shape ``(T, H, W, 3)``.
 22        fps (float): Output video frame rate.
 23
 24    Raises:
 25        ValueError: If ``frames`` does not have shape ``(T, H, W, 3)`` or
 26            contains zero timesteps.
 27    """
 28    if frames.ndim != 4 or frames.shape[-1] != 3:
 29        raise ValueError(f"Expected frames with shape (T,H,W,3), got {frames.shape}")
 30    if frames.shape[0] == 0:
 31        raise ValueError("Cannot write video with zero frames")
 32
 33    with imageio.get_writer(
 34        str(video_path),
 35        format="FFMPEG",
 36        mode="I",
 37        fps=float(fps),
 38        codec="libx264",
 39    ) as writer:
 40        for frame in frames:
 41            writer.append_data(np.asarray(frame, dtype=np.uint8))

 44class EpisodeRecorder:
 45    """Record rollout observations and persist them in HDF5 format.
 46
 47    The recorder stores proprioception, optional video references, and
 48    annotation metadata under the robotic failure upload schema.
 49    """
 50
 51    def __init__(
 52        self,
 53        output_dir: str,
 54        policy_name: str,
 55        policy_id: str = "A",
 56        session_id: str | None = None,
 57        camera_names: list[str] | None = None,
 58        num_cameras: int | None = None,
 59        robot_id: str = "",
 60        control_freq: str = "",
 61    ) -> None:
 62        """Initialize a recorder instance.
 63
 64        Args:
 65            output_dir (str): Base output directory for saved artifacts.
 66            policy_name (str): Name of the policy being evaluated.
 67            policy_id (str): Policy identifier for metadata and filenames.
 68            session_id (str | None): Optional unique session ID; generated when
 69                omitted.
 70            camera_names (list[str] | None): Optional explicit camera names.
 71            num_cameras (int | None): Number of cameras used when
 72                ``camera_names`` is not set.
 73            robot_id (str): Optional robot ID stored in episode annotations.
 74            control_freq (str): Optional controller frequency metadata.
 75
 76        Raises:
 77            ValueError: If ``num_cameras`` is negative.
 78        """
 79        self.output_dir = Path(output_dir)
 80        self.policy_name = policy_name
 81        self.policy_id = policy_id
 82        self.session_id = session_id or str(uuid.uuid4())
 83        if camera_names is not None:
 84            self.camera_names = list(camera_names)
 85        elif num_cameras is not None:
 86            if num_cameras < 0:
 87                raise ValueError("num_cameras must be >= 0")
 88            self.camera_names = [f"camera_{idx}" for idx in range(num_cameras)]
 89        else:
 90            self.camera_names = []
 91        self.robot_id = robot_id
 92        self.control_freq = str(control_freq) if control_freq else ""
 93
 94        # Per-camera frame buffers keyed by camera name
 95        self.frames: dict[str, list[np.ndarray]] = {cam: [] for cam in self.camera_names}
 96
 97        # Timestep data buffer
 98        self.timesteps: list[dict[str, Any]] = []

100    def record_step(self, observation: dict[str, Any], action: np.ndarray) -> None:
101        """Append one rollout timestep to in-memory buffers.
102
103        Args:
104            observation (dict[str, Any]): Observation payload containing state
105                and optional images.
106            action (np.ndarray): Action vector applied at this timestep.
107
108        Returns:
109            None: This method only updates in-memory buffers.
110        """
111        action_arr = np.asarray(action, dtype=np.float32)
112
113        # Buffer frames for each configured camera (if available)
114        for cam in self.camera_names:
115            frame = self._get_camera_frame(observation, cam)
116            if frame is not None:
117                self.frames[cam].append(np.asarray(frame, dtype=np.uint8))
118
119        if action_arr.shape[0] >= 1:
120            gripper_position_cmd = np.asarray([action_arr[-1]], dtype=np.float32)
121        else:
122            gripper_position_cmd = np.zeros((1,), dtype=np.float32)
123
124        gripper_velocity_cmd = np.asarray(
125            observation.get("gripper_velocity", np.zeros((1,), dtype=np.float32)),
126            dtype=np.float32,
127        )
128        eef_cartesian_position_cmd = np.asarray(
129            observation.get(
130                "eef_cartesian_position",
131                observation.get("cartesian_position_command", np.full((6,), np.nan, dtype=np.float32)),
132            ),
133            dtype=np.float32,
134        )
135        eef_cartesian_velocity_cmd = np.asarray(
136            observation.get(
137                "eef_cartesian_velocity",
138                observation.get("cartesian_velocity_command", np.full((6,), np.nan, dtype=np.float32)),
139            ),
140            dtype=np.float32,
141        )
142
143        # Buffer timestep data
144        self.timesteps.append(
145            {
146                "cartesian_position": np.asarray(
147                    observation["cartesian_position"], dtype=np.float32
148                ),
149                "joint_position": np.asarray(
150                    observation["joint_position"], dtype=np.float32
151                ),
152                "gripper_position": np.asarray(
153                    observation["gripper_position"], dtype=np.float32
154                ),
155                "action_dict": {
156                    "gripper_position": gripper_position_cmd,
157                    "gripper_velocity": gripper_velocity_cmd,
158                    "eef_cartesian_position": eef_cartesian_position_cmd,
159                    "eef_cartesian_velocity": eef_cartesian_velocity_cmd,
160                },
161            }
162        )

164    def save(self, metadata: dict[str, Any]) -> Path:
165        """Persist the currently buffered episode to disk.
166
167        Args:
168            metadata (dict[str, Any]): Save metadata containing language and
169                annotation fields.
170
171        Returns:
172            Path: Path to the written HDF5 episode file.
173
174        Raises:
175            ValueError: If no rollout steps were recorded.
176        """
177        if len(self.timesteps) == 0:
178            raise ValueError("No steps recorded. Call record_step() first.")
179
180        normalized_metadata = self._normalize_metadata(metadata)
181
182        # Save all artifacts flat in output_dir
183        self.output_dir.mkdir(parents=True, exist_ok=True)
184
185        # Resolve and/or save per-camera MP4s, then persist their paths in HDF5.
186        # All files are written directly into output_dir.
187        resolved_metadata = dict(normalized_metadata)
188        resolved_metadata["video_paths"] = self._resolve_video_paths(
189            output_dir=self.output_dir,
190            provided_video_paths=normalized_metadata.get("video_paths", {}),
191        )
192
193        # Save HDF5 file
194        h5_filename = f"{self.session_id}_{self.policy_id}_{self.policy_name}.h5"
195        h5_path = self.output_dir / h5_filename
196        self._save_h5(h5_path, resolved_metadata)
197
198        return h5_path

200    def _save_h5(self, path: Path, metadata: dict[str, Any]) -> None:
201        """Write buffered rollout data and metadata into one HDF5 file.
202
203        Args:
204            path (Path): Target HDF5 file path.
205            metadata (dict[str, Any]): Normalized metadata payload.
206
207        Returns:
208            None: This method only performs file I/O side effects.
209        """
210        str_dtype = h5py.string_dtype(encoding="utf-8")
211        with h5py.File(path, "w") as f:
212            f.attrs["schema"] = "robotic_failure_upload_data_format_v1"
213
214            ann_in = metadata.get("episode_annotations", {})
215            if not isinstance(ann_in, dict):
216                ann_in = {}
217            episode_annotations = f.create_group("episode_annotations")
218
219            # Fixed metadata fields stored as attributes
220            for key in ("episode_id", "lab_id", "operator_name", "policy_id", "robot_id", "control_freq"):
221                value = ann_in.get(key)
222                if value is not None:
223                    episode_annotations.attrs[key] = str(value)
224
225            success = ann_in.get("success")
226            if success is not None:
227                episode_annotations.attrs["success"] = float(np.float32(success))
228
229            # Questionnaire result stored as a dataset (variable-length JSON)
230            failure_annotation_in = ann_in.get("failure_annotation")
231            if failure_annotation_in is not None:
232                failure_annotation = episode_annotations.create_group("failure_annotation")
233                if isinstance(failure_annotation_in, dict):
234                    if set(failure_annotation_in.keys()) == {"question"}:
235                        question_value = str(failure_annotation_in.get("question", ""))
236                    else:
237                        question_value = json.dumps(failure_annotation_in, ensure_ascii=False)
238                else:
239                    question_value = str(failure_annotation_in)
240                failure_annotation.create_dataset("question", data=question_value, dtype=str_dtype)
241
242            # Store per-camera MP4 file paths (strings), not inlined frame tensors.
243            image_group = f.create_group("image_observations")
244            video_paths = metadata.get("video_paths", {})
245            if not isinstance(video_paths, dict):
246                video_paths = {}
247            for cam in self.camera_names:
248                raw_video_path = str(video_paths.get(cam, "")).strip()
249                if not raw_video_path:
250                    continue
251
252                video_path_obj = Path(raw_video_path).expanduser()
253                if video_path_obj.is_absolute():
254                    rel_video_path = os.path.relpath(
255                        video_path_obj.resolve(),
256                        start=path.parent.resolve(),
257                    )
258                else:
259                    rel_video_path = os.path.relpath(
260                        (path.parent / video_path_obj).resolve(),
261                        start=path.parent.resolve(),
262                    )
263
264                dataset = image_group.create_dataset(
265                    cam,
266                    data=rel_video_path.replace(os.sep, "/"),
267                    dtype=str_dtype,
268                )
269                dataset.attrs["format"] = "mp4_filepath"
270
271            observation_group = f.create_group("observation")
272            observation_group.create_dataset(
273                "gripper_position",
274                data=np.stack([t["gripper_position"] for t in self.timesteps], axis=0),
275                dtype=np.float64,
276            )
277            observation_group.create_dataset(
278                "eef_cartesian_position",
279                data=np.stack([t["cartesian_position"] for t in self.timesteps], axis=0),
280                dtype=np.float64,
281            )
282            observation_group.create_dataset(
283                "joint_position",
284                data=np.stack([t["joint_position"] for t in self.timesteps], axis=0),
285                dtype=np.float64,
286            )
287
288            action_group = f.create_group("action_dict")
289            action_group.create_dataset(
290                "gripper_position",
291                data=np.stack([t["action_dict"]["gripper_position"] for t in self.timesteps], axis=0),
292                dtype=np.float64,
293            )
294            action_group.create_dataset(
295                "gripper_velocity",
296                data=np.stack([t["action_dict"]["gripper_velocity"] for t in self.timesteps], axis=0),
297                dtype=np.float64,
298            )
299            action_group.create_dataset(
300                "eef_cartesian_position",
301                data=np.stack([t["action_dict"]["eef_cartesian_position"] for t in self.timesteps], axis=0),
302                dtype=np.float64,
303            )
304            action_group.create_dataset(
305                "eef_cartesian_velocity",
306                data=np.stack([t["action_dict"]["eef_cartesian_velocity"] for t in self.timesteps], axis=0),
307                dtype=np.float64,
308            )
309
310            f.attrs["language_instruction"] = str(metadata.get("language_instruction", ""))

312    def _normalize_metadata(self, metadata: dict[str, Any] | None) -> dict[str, Any]:
313        """Validate and normalize user-provided metadata.
314
315        Args:
316            metadata (dict[str, Any] | None): Raw metadata dictionary or
317                ``None``.
318
319        Returns:
320            dict[str, Any]: Normalized metadata containing allowed top-level
321                keys only.
322
323        Raises:
324            TypeError: If metadata or nested fields have invalid types.
325            ValueError: If unsupported metadata keys are present.
326        """
327        if metadata is None:
328            metadata = {}
329        if not isinstance(metadata, dict):
330            raise TypeError("metadata must be a dict")
331
332        allowed_metadata_keys = {"language_instruction", "episode_annotations", "video_paths"}
333        unknown_metadata_keys = set(metadata.keys()) - allowed_metadata_keys
334        if unknown_metadata_keys:
335            raise ValueError(
336                "Unsupported metadata keys: "
337                + ", ".join(sorted(unknown_metadata_keys))
338                + ". Allowed keys: language_instruction, episode_annotations, video_paths"
339            )
340
341        episode_annotations = metadata.get("episode_annotations", {})
342        if episode_annotations is None:
343            episode_annotations = {}
344        if not isinstance(episode_annotations, dict):
345            raise TypeError("metadata['episode_annotations'] must be a dict when provided")
346
347        allowed_annotation_keys = {
348            "episode_id",
349            "lab_id",
350            "operator_name",
351            "policy_id",
352            "robot_id",
353            "control_freq",
354            "success",
355            "failure_annotation",
356        }
357        unknown_annotation_keys = set(episode_annotations.keys()) - allowed_annotation_keys
358        if unknown_annotation_keys:
359            raise ValueError(
360                "Unsupported episode_annotations keys: "
361                + ", ".join(sorted(unknown_annotation_keys))
362                + ". Allowed keys: "
363                + ", ".join(sorted(allowed_annotation_keys))
364            )
365
366        failure_annotation = episode_annotations.get("failure_annotation")
367        if failure_annotation is not None and not isinstance(failure_annotation, (dict, str)):
368            raise TypeError(
369                "metadata['episode_annotations']['failure_annotation'] must be a dict or str"
370            )
371
372        video_paths = metadata.get("video_paths", {})
373        if video_paths is None:
374            video_paths = {}
375        if not isinstance(video_paths, dict):
376            raise TypeError("metadata['video_paths'] must be a dict when provided")
377
378        return {
379            "language_instruction": str(metadata.get("language_instruction", "")),
380            "episode_annotations": dict(episode_annotations),
381            "video_paths": dict(video_paths),
382        }

384    def _resolve_video_paths(
385        self,
386        output_dir: Path,
387        provided_video_paths: Any,
388    ) -> dict[str, str]:
389        """Resolve per-camera MP4 paths, writing videos when needed.
390
391        Args:
392            output_dir (Path): Directory used as the base for relative path
393                storage.
394            provided_video_paths (Any): Optional external mapping from camera to
395                MP4 path.
396
397        Returns:
398            dict[str, str]: Mapping from camera name to relative MP4 path.
399        """
400        paths: dict[str, str] = {}
401        provided = provided_video_paths if isinstance(provided_video_paths, dict) else {}
402
403        for cam in self.camera_names:
404            provided_path = str(provided.get(cam, ""))
405            if provided_path:
406                abs_path = Path(provided_path).expanduser().resolve()
407                if abs_path.suffix.lower() != ".mp4":
408                    abs_path = abs_path.with_suffix(".mp4")
409                paths[cam] = os.path.relpath(abs_path, start=output_dir)
410                continue
411
412            frames = self.frames.get(cam, [])
413            if len(frames) == 0:
414                continue
415
416            video_path = output_dir / f"{self.session_id}_{self.policy_id}_{self.policy_name}_{cam}.mp4"
417            write_mp4(video_path=video_path, frames=np.asarray(frames), fps=10.0)
418            paths[cam] = os.path.relpath(video_path.resolve(), start=output_dir)
419
420        return paths

422    def _get_camera_frame(self, observation: dict[str, Any], cam_name: str) -> np.ndarray | None:
423        """Extract a camera frame from supported observation key patterns.
424
425        Args:
426            observation (dict[str, Any]): Observation dictionary for one
427                timestep.
428            cam_name (str): Canonical camera name to resolve.
429
430        Returns:
431            np.ndarray | None: Camera frame array when available, otherwise
432                ``None``.
433        """
434        candidates = (cam_name, f"image_{cam_name}", f"{cam_name}_image")
435        for key in candidates:
436            if key in observation:
437                return np.asarray(observation[key])
438        return None

491    def prompt(self, language_instruction: str = "") -> dict[str, Any]:
492        """Prompt for annotation fields and return recorder-ready metadata.
493
494        Args:
495            language_instruction (str): Optional pre-filled instruction string.
496
497        Returns:
498            dict[str, Any]: Metadata dictionary compatible with
499                ``EpisodeRecorder.save``.
500        """
501        print("\n" + "=" * 50)
502        print("Episode Annotation")
503        print("=" * 50)
504
505        # Language instruction
506        if language_instruction:

508            edit = input("Edit instruction? [y/N]: ").strip().lower()
509            if edit == "y":
510                language_instruction = input("New instruction: ").strip()
511        else:
512            language_instruction = input("Language instruction: ").strip()
513
514        # Success
515        while True:
516            success_input = input("\nSuccess score (0.0-1.0): ").strip()
517            try:
518                success = float(success_input)
519                if 0.0 <= success <= 1.0:
520                    break
521                print("Please enter a value between 0.0 and 1.0")
522            except ValueError:
523                print("Please enter a valid number")
524
525        # Questionnaire reply payload
526        print("\nQuestionnaire response (describe what happened, press Enter twice to finish):")
527        lines = []
528        while True:
529            line = input()
530            if line == "":
531                break
532            lines.append(line)
533        question = "\n".join(lines)
534
535        print("=" * 50 + "\n")
536
537        return {
538            "language_instruction": language_instruction,
539            "episode_annotations": {
540                "success": success,
541                "failure_annotation": {
542                    "question": question,
543                },
544            },
545        }
