robotic_failure_data.annotation_tool.rollout_annotator

Bridges the droid inference loop with the browser-based annotation tool.

After each rollout, RolloutAnnotator saves the camera MP4s and JSON metadata to the samples directory, starts the annotation Flask server if needed, signals the task page to navigate to the annotation view, and optionally blocks until the annotation is submitted.

140    def start(self) -> None:
141        """Start or connect to the annotation server and wait for session setup.
142
143        Returns:
144            None: This method runs startup side effects only.
145        """
146        self.samples_dir.mkdir(parents=True, exist_ok=True)
147        if self._is_port_open():
148            print(f"Annotation server already running on port {self.annotator_port}.")
149        else:
150            print(f"Starting annotation server on port {self.annotator_port}...")
151            self._proc = subprocess.Popen(
152                [
153                    "uv", "run", "python", "-m", "robotic_failure_data.annotation_tool.annotator",
154                    "--samples-dir", str(self.samples_dir),
155                    "--port", str(self.annotator_port),
156                    "--no-browser",
157                ],
158                stdout=subprocess.DEVNULL,
159                stderr=subprocess.DEVNULL,
160            )
161            atexit.register(self._stop)
162
163            for _ in range(20):
164                time.sleep(0.5)
165                if self._is_port_open():
166                    print("Annotation server ready.")
167                    break
168            else:
169                print("Warning: annotation server did not start in time. Continuing anyway.")
170
171        # Prefill the session form with script-known values
172        try:
173            self._api_post("/api/session/prefill", {
174                "policy_id": self.policy_id,
175                "camera_names": self.camera_names,
176            })
177        except Exception:
178            pass
179
180        webbrowser.open(f"http://localhost:{self.annotator_port}/task")
181
182        # Block until the user configures the session in the browser
183        print("Waiting for session configuration in browser...")
184        while True:
185            try:
186                session = self._api_get("/api/session")
187                if session.get("configured"):
188                    self._session_dir = Path(session["session_dir"])
189                    self.camera_names = session.get("camera_names", self.camera_names)
190                    self._session_annotator_name = session.get("annotator_name", "")
191                    print(f"Session configured. Samples dir: {self._session_dir}")
192                    break
193            except Exception:
194                pass
195            time.sleep(0.5)
196

199
200        Returns:
201            None: This method runs shutdown side effects only.
202        """

204            self._proc.terminate()

274    def finish_rollout(
275        self,
276        instruction: str,
277        videos: dict[str, list[np.ndarray]],
278        t_step: int,
279        recorder: EpisodeRecorder | None = None,
280    ) -> float | None:
281        """Persist rollout artifacts, request annotation, and save HDF5 data.
282
283        Args:
284            instruction (str): Language instruction executed by the policy.
285            videos (dict[str, list[np.ndarray]]): Per-camera frame sequences
286                keyed by camera name.
287            t_step (int): Final rollout timestep index.
288            recorder (EpisodeRecorder | None): Optional external recorder to use
289                instead of internal one.
290
291        Returns:
292            float | None: ``1.0`` for success, ``0.0`` for failure, or ``None``
293                when ``wait_for_annotation`` is ``False``.
294        """
295        active_recorder = recorder or self._active_recorder
296        annotation: dict[str, Any] = {}
297
298        sample_id, video_paths = self._save_sample(instruction, videos, t_step)
299        self._last_sample_id = sample_id
300
301        # Signal the task page to redirect to the annotation view
302        samples_listing = self._api_get("/api/samples")
303        idx = 0
304        if isinstance(samples_listing, list) and len(samples_listing) > 0:
305            idx = next(
306                (
307                    i
308                    for i, sample in enumerate(samples_listing)
309                    if isinstance(sample, dict) and sample.get("id") == sample_id
310                ),
311                len(samples_listing) - 1,
312            )
313        self._api_post("/api/task/annotating", {"sample_idx": idx})
314
315        success: float | None = None
316        if self.wait_for_annotation:
317            annotation = self._wait_for_annotation(sample_id)
318            success = 1.0 if annotation.get("binary_success") == "Success" else 0.0
319            print(f"Annotation received: {annotation.get('binary_success', '?')}")
320
321            # Write annotation back into the sample JSON so it's co-located with the videos.
322            # Use an atomic write (temp file + rename) to avoid racing with the Flask server.
323            session_dir = self._session_dir or self.samples_dir
324            sample_json = session_dir / f"{sample_id}.json"
325            if sample_json.exists():
326                metadata = json.loads(sample_json.read_text())
327                metadata["annotation"] = {k: v for k, v in annotation.items()
328                                           if k not in {"annotated_at", "annotator", "source"}}
329                fd, tmp = tempfile.mkstemp(dir=session_dir, suffix=".json.tmp")
330                try:
331                    with os.fdopen(fd, "w") as f:
332                        json.dump(metadata, f, indent=2)
333                    os.replace(tmp, sample_json)
334                except Exception:
335                    os.unlink(tmp)
336                    raise
337
338        # Reset task page to idle so the user can submit the next task
339        self._api_post("/api/task/done", {})
340
341        if active_recorder is not None and active_recorder.num_steps > 0:
342            episode_annotations = {
343                "policy_id": self.policy_id,
344                "success": success if success is not None else 0.0,
345                "failure_annotation": annotation if self.wait_for_annotation else {},
346            }
347            if self._session_annotator_name:
348                episode_annotations["operator_name"] = self._session_annotator_name
349
350            active_recorder.save({
351                "language_instruction": instruction,
352                "episode_annotations": episode_annotations,
353                "video_paths": video_paths,
354            })
355        self._active_recorder = None
356
357        return success

363    def _save_sample(
364        self,
365        instruction: str,
366        videos: dict[str, list[np.ndarray]],
367        t_step: int,
368    ) -> tuple[str, dict[str, str]]:
369        """Write rollout videos and JSON metadata into the session directory.
370
371        Args:
372            instruction (str): Language instruction associated with the rollout.
373            videos (dict[str, list[np.ndarray]]): Per-camera frame sequences
374                keyed by camera name.
375            t_step (int): Final rollout timestep index.
376
377        Returns:
378            tuple[str, dict[str, str]]: Tuple of ``(sample_id, video_paths)``
379                where ``video_paths`` maps camera name to absolute MP4 path.
380        """
381        sample_id = datetime.datetime.now().strftime("%Y%m%d_%H%M%S") + f"_{self.policy_name}"
382        session_dir = self._session_dir or self.samples_dir
383        session_dir.mkdir(parents=True, exist_ok=True)
384        video_paths: dict[str, str] = {}
385
386        print(f"Saving videos to {session_dir} ...")
387        for cam_name, frames in videos.items():
388            video_path = session_dir / f"{sample_id}_{cam_name}.mp4"
389            ImageSequenceClip(list(np.stack(frames)), fps=10).write_videofile(
390                str(video_path),
391                codec="libx264",
392                logger=None,
393            )
394            video_paths[cam_name] = str(video_path.resolve())
395
396        (session_dir / f"{sample_id}.json").write_text(
397            json.dumps({
398                "language_instruction": instruction,
399                "policy_name": self.policy_name,
400                "policy_id": self.policy_id,
401                "camera_names": list(videos.keys()),
402                "num_steps": t_step + 1,
403                "timestamp": datetime.datetime.now().isoformat(),
404            })
405        )
406        return sample_id, video_paths

408    def _wait_for_annotation(self, sample_id: str) -> dict[str, Any]:
409        """Wait until a non-VLM annotation exists for the given sample.
410
411        Args:
412            sample_id (str): Sample identifier to monitor in
413                ``annotations.json``.
414
415        Returns:
416            dict[str, Any]: Annotation dictionary provided by a human annotator.
417        """
418        print("Waiting for annotation in browser... (annotate the video and click Save)")
419        while True:
420            try:
421                data = self._api_get(f"/api/annotations/{sample_id}")
422                if isinstance(data, dict):
423                    for ann in data.values():
424                        if isinstance(ann, dict) and ann.get("source") != "vlm":
425                            return ann
426            except Exception:
427                pass
428            time.sleep(1)

534    def finish_rollout(
535        self,
536        instruction: str,
537        videos: dict[str, list[np.ndarray]],
538        t_step: int,
539        recorder: EpisodeRecorder | None = None,
540    ) -> float:
541        """Collect annotation answers in CLI mode and persist episode metadata.
542
543        Args:
544            instruction (str): Language instruction executed by the policy.
545            videos (dict[str, list[np.ndarray]]): Per-camera frame sequences
546                keyed by camera name.
547            t_step (int): Final rollout timestep index.
548            recorder (EpisodeRecorder | None): Optional external recorder to use
549                instead of internal one.
550
551        Returns:
552            float: ``1.0`` when binary success is ``"Success"`` and ``0.0``
553                otherwise.
554        """
555        active_recorder = recorder or self._active_recorder
556        self._last_instruction = instruction
557        annotation = self._prompt_questionnaire()
558        annotation.update({
559            "annotated_at": datetime.datetime.now().isoformat(),
560            "annotator": self._annotator_name,
561            "source": "human",
562        })
563
564        self._last_sample_id = (
565            datetime.datetime.now().strftime("%Y%m%d_%H%M%S") + f"_{self.policy_name}"
566        )
567
568        # Save to annotations.json in the same format as the web tool
569        annotations = json.loads(ANNOTATIONS_FILE.read_text()) if ANNOTATIONS_FILE.exists() else {}
570        annotations.setdefault(self._last_sample_id, {})[self._annotator_name] = annotation
571        ANNOTATIONS_FILE.write_text(json.dumps(annotations, indent=2, sort_keys=True))
572
573        success = 1.0 if annotation.get("binary_success") == "Success" else 0.0
574
575        if active_recorder is not None and active_recorder.num_steps > 0:
576            active_recorder.save({
577                "language_instruction": instruction,
578                "episode_annotations": {
579                    "policy_id": self.policy_id,
580                    "operator_name": self._annotator_name,
581                    "success": success,
582                    "failure_annotation": annotation,
583                },
584            })
585
586        self._active_recorder = None
587
588        return success

606    # Questions that are only relevant on failure
607    _FAILURE_ONLY_QUESTIONS = {"failure_category", "failure_description", "would_retry"}

609    def _prompt_questionnaire(self) -> dict[str, Any]:
610        """Prompt all relevant questionnaire entries and collect answers.
611
612        Returns:
613            dict[str, Any]: Mapping from question IDs to user answers.
614        """
615        print("\n--- Annotation ---")
616        answers: dict[str, Any] = {}
617        binary_success: Any = None
618        for q in self._questionnaire.get("questions", []):
619            if binary_success == "Success" and q["id"] in self._FAILURE_ONLY_QUESTIONS:
620                continue
621            answer = self._prompt_question(q)
622            if answer is not None:
623                answers[q["id"]] = answer
624            if q["id"] == "binary_success":
625                binary_success = answer
626        return answers

628    def _prompt_question(self, q: dict[str, Any]) -> str | list[str] | None:
629        """Prompt for one questionnaire item and validate the response.
630
631        Args:
632            q (dict[str, Any]): Questionnaire item definition with ``id``,
633                ``type``, and ``label``.
634
635        Returns:
636            str | list[str] | None: Answer value as a string, list of strings
637                for checkbox questions, or ``None`` when an optional question is
638                skipped.
639        """
640        required = q.get("required", False)
641        question_type = q["type"]
642        label = q["label"]
643        options = q.get("options", [])
644        skip_hint = "" if required else " (Enter to skip)"
645
646        while True:
647            if question_type in ("radio", "select"):
648                print(f"\n{label}")
649                for i, opt in enumerate(options, 1):
650                    print(f"  {i}. {opt}")
651                raw = input(f"Choice [1-{len(options)}]{skip_hint}: ").strip()
652                if not raw and not required:
653                    return None
654                try:
655                    idx = int(raw) - 1
656                    if 0 <= idx < len(options):
657                        return options[idx]
658                except ValueError:
659                    if raw in options:
660                        return raw
661                print(f"  Please enter a number between 1 and {len(options)}.")
662
663            elif question_type == "checkbox":
